No Refuge: A Decade of Australian Housing

A light curiosity using AURIN-Domain property data.

  • 📖 Story
  • 💻 The Code

What separates a city where property doubles from one where it barely moves?

In 2016, Brisbane’s median house price was almost 50% lower than Melbourne’s. Ten years later, Brisbane’s median price had more than doubled, while Melbourne’s had seen only modest growth. By 2026, Brisbane had overtaken Melbourne.

For the buyer who owned in Brisbane in 2016, the last decade produced extraordinary wealth. For a 25-year-old trying to buy their first home there in 2026 though, the same numbers describe a significant barrier that did not exist when their parents or older siblings entered the market. The data tells both stories simultaneously.

In what follows, data from the Domain property listings platform, accessed via AURIN, will be used to answer two important questions:

  1. How did Australian cities diverge after 2020? We use the Domain’s suburb performance statistics endpoint and showcase how a small number of API calls returns a decade of quarterly data for Australian cities (Chapter 1).

  2. Is Melbourne’s flatness really flat, or is something more interesting happening inside it? We use the Domain’s residential listings endpoint to look at individual transactions. The aggregate median hides a within-city story that only point-level data can reveal. Chapter 2.

# ── Manifest helpers (resume-safe: already-downloaded files are skipped) ────

def _load_manifest():
    return json.loads(_MANIFEST.read_text()) if _MANIFEST.exists() else {"captures": []}

def _save_manifest(m):
    _MANIFEST.write_text(json.dumps(m, indent=2))

def _already_captured(m, rel):
    return str(rel) in {e["file"] for e in m["captures"]}

def _record_capture(m, rel, endpoint, params, status, section, empty=False):
    m["captures"].append({
        "file":             str(rel),
        "endpoint":         endpoint,
        "params":           params,
        "captured_at":      datetime.now(timezone.utc).isoformat(),
        "http_status":      status,
        "empty_response":   empty,
        "datatale_section": section,
    })
    _save_manifest(m)

def _save_json(rel, data):
    dest = _DATATALE / rel
    dest.parent.mkdir(parents=True, exist_ok=True)
    dest.write_text(json.dumps(data, indent=2))

def _save_probe(rel, response):
    _save_json(rel, {
        "x_total_count": int(response.headers.get("X-Total-Count", 0)),
        "body":          response.json() if response.text.strip() else {},
    })

def _extract_listings(data):
    if not isinstance(data, list):
        return []
    out = []
    for item in data:
        raw = item.get("listing") or item.get("listings")
        if isinstance(raw, dict):
            out.append(raw)
        elif isinstance(raw, list):
            out.extend(raw)
    return out


def _fetch_listings_pages(tracker, manifest, suburb, postcode, state,
                          start_date, out_dir, datatale_section):
    out_abs    = _DATATALE / out_dir
    state_file = out_abs / "fetch_state.json"
    page_size  = 200

    if state_file.exists():
        st       = json.loads(state_file.read_text())
        cursor   = st["cursor"]
        run      = st["run"]
        seen_ids = set(st["seen_ids"])
        print(f"  Resuming: cursor={cursor}  completed runs={run}  seen={len(seen_ids)}")
    else:
        cursor   = start_date
        run      = 0
        seen_ids = set()

    probe_path = f"{out_dir}/probe.json"
    if not _already_captured(manifest, probe_path):
        probe_r = tracker.post(
            f"{PROXY_BASE}/v1/listings/residential/_search",
            json_body={
                "listingType": "Sold",
                "listedSince": start_date,
                "sort": {"sortKey": "DateListed", "direction": "Ascending"},
                "locations": [{"state": state, "suburb": suburb,
                               "postCode": postcode, "includeSurroundingSuburbs": False}],
                "pageSize": 1, "pageNumber": 1,
            },
        )
        _save_probe(probe_path, probe_r)
        total_count = int(probe_r.headers.get("X-Total-Count", 0))
        _record_capture(manifest, probe_path,
                        endpoint="POST /v1/listings/residential/_search",
                        params={"suburb": suburb, "postcode": postcode,
                                "listedSince": start_date},
                        status=probe_r.status_code,
                        section=datatale_section)
        print(f"  Probe: {total_count:,} total records for {suburb}")
    else:
        probe_data  = json.loads((_DATATALE / probe_path).read_text())
        total_count = probe_data.get("x_total_count", 0)
        print(f"  Probe already captured: {total_count:,} records")

    if total_count == 0:
        print(f"  WARNING: probe returned 0 -- check suburb name and postcode")
        return

    payload = {
        "listingType": "Sold",
        "sort":        {"sortKey": "DateListed", "direction": "Ascending"},
        "locations":   [{"state": state, "suburb": suburb,
                          "postCode": postcode, "includeSurroundingSuburbs": False}],
    }

    while True:
        run      += 1
        page      = 1
        run_batch = []
        hit_limit = False
        fetch_err = False

        while True:
            page_path = f"{out_dir}/run_{run:03d}_page_{page:03d}.json"

            if _already_captured(manifest, page_path):
                existing = json.loads((_DATATALE / page_path).read_text())
                run_batch.extend(_extract_listings(existing))
                print(f"    run {run:03d} page {page:03d} -- already captured, skip")
                page += 1
                continue

            r = tracker.post(
                f"{PROXY_BASE}/v1/listings/residential/_search",
                json_body={**payload, "listedSince": cursor,
                           "pageSize": page_size, "pageNumber": page},
            )

            if r.status_code == 400:
                hit_limit = True
                break
            if r.status_code != 200:
                print(f"    ERROR {r.status_code} run {run:03d} page {page:03d}: {r.text[:100]}")
                print(f"  Stopping -- re-run cell to resume from cursor {cursor}")
                fetch_err = True
                break

            data     = r.json()
            listings = _extract_listings(data)
            new      = [l for l in listings if l.get("id") and l["id"] not in seen_ids]

            _save_json(page_path, data)
            _record_capture(manifest, page_path,
                            endpoint="POST /v1/listings/residential/_search",
                            params={**payload, "listedSince": cursor,
                                    "pageSize": page_size, "pageNumber": page},
                            status=r.status_code,
                            section=datatale_section)
            run_batch.extend(listings)
            seen_ids.update(l["id"] for l in listings if l.get("id"))
            print(f"    run {run:03d} page {page:03d}: {len(listings)} records  "
                  f"({len(new)} new  {len(seen_ids)} unique total)  [{tracker.total} credits]")

            if len(listings) < page_size:
                break
            page += 1

        if fetch_err:
            break

        if not hit_limit:
            print(f"  Done. {len(seen_ids):,} unique listings for {suburb}.")
            break

        if not run_batch:
            print("  Empty batch after 400 -- stopping")
            break

        dates = [l.get("dateListed", "") for l in run_batch if l.get("dateListed")]
        if not dates:
            print("  No dateListed in batch -- cannot advance cursor, stopping")
            break
        new_cursor = max(dates)[:10]
        if new_cursor == cursor:
            print(f"  Cursor stuck at {cursor} (all same date) -- stopping")
            break

        print(f"  Limit hit. Advancing cursor {cursor} -> {new_cursor}")
        cursor = new_cursor
        state_file.write_text(json.dumps(
            {"cursor": cursor, "run": run, "seen_ids": list(seen_ids)}, indent=2,
        ))

Before We Begin: Setting Up the Analysis

The code cells below prepare everything needed before the data exploration starts. Here is what each step does.

Libraries: we load standard Python tools for data work and visualisation.

  • numpy and pandas handle numerical operations and tabular data.
  • plotly creates the interactive charts throughout the notebook.
  • folium builds the interactive geographic map in Chapter 2.
  • The built-in json and pathlib modules read data files from disk.

Directory paths and study window: we point Python at the two data folders on disk (statistics/ for quarterly suburb aggregates, listings/ for individual property transactions) and define a study window from 2018 to 2026. Q1 2018 is the index baseline used throughout Chapter 1.

Two helper functions:

  • load_stats() reads a single statistics JSON file and returns a tidy table with one row per quarter.

  • pct_change_from_q1_2018() converts raw dollar prices into the index used throughout Chapter 1, where every city starts at zero in Q1 2018 and subsequent values show the percentage gain or loss from that point.

City colours and labels: a consistent colour is assigned to each city (Melbourne is always blue, Brisbane always green, and so on) so every chart in the notebook uses the same visual language. A matching dictionary maps internal city keys (like "sunshine_coast") to the display names that appear in chart titles and tooltips.

def load_stats(path):
    """Read one suburb/category statistics JSON file and return a tidy DataFrame.

    The Domain API returns a nested 'series' > 'seriesInfo' structure; this
    function flattens it into one row per quarter.

    Returns None if the file is empty, missing the 'series' key, or contains
    no quarters with a medianSoldPrice (i.e. the suburb had too few sales to
    report in any quarter).

    Columns in the returned DataFrame (date-indexed, sorted ascending):
        medianSoldPrice, daysOnMarket, numberSold, p25, p75, meanRentYield
    """
    data = json.loads(path.read_text())
    if not data or "series" not in data:
        return None
    rows = data["series"]["seriesInfo"]
    records = [
        {
            "date":            pd.Timestamp(year=r["year"], month=r["month"], day=1),
            "medianSoldPrice": r["values"].get("medianSoldPrice"),
            "daysOnMarket":    r["values"].get("daysOnMarket"),
            "numberSold":      r["values"].get("numberSold"),
            "p25":             r["values"].get("25thPercentileSoldPrice"),
            "p75":             r["values"].get("75thPercentileSoldPrice"),
            "meanRentYield":   r["values"].get("meanRentYield"),
        }
        for r in rows
        if r["values"].get("medianSoldPrice")   # skip quarters with no price data
    ]
    if not records:
        return None
    return pd.DataFrame(records).set_index("date").sort_index()


def pct_change_from_q1_2018(df, col="medianSoldPrice"):
    """Add an 'indexed' column expressing price as % change from the Q1 2018 level.

    Q1 2018 is defined as any record where year==2018 and month in {1, 2, 3}.
    The baseline is the first available price in that window.

    Examples:
        indexed = +80  ->  price is 80% above its Q1 2018 level
        indexed =  -5  ->  price is 5% below its Q1 2018 level
        indexed =   0  ->  price equals its Q1 2018 level (the baseline quarter)

    Returns df unchanged (without an 'indexed' column) if no Q1 2018 data exists.
    """
    mask     = (df.index.year == 2018) & (df.index.month.isin([1, 2, 3]))
    baseline = df.loc[mask, col].dropna()
    if baseline.empty:
        return df
    df = df.copy()
    df["indexed"] = (df[col] / baseline.iloc[0] - 1) * 100   # 0 = Q1 2018 level
    return df

In 2018, Melbourne’s median house sat nearly 50% above Brisbane’s. That gap held through 2018. Then rates went to zero, then shot back up, and five years passed.

The charts below index every city to Q1 2018 = 0%, the start of our study period. A reading of +80 means prices are 80% above that baseline; -5 means they have softened below it.

The median (the middle sale, not the average) is used throughout to limit distortion from a small number of very expensive transactions.

The map below shows cumulative growth since Q1 2018 for every major city. Bigger and redder means more growth.

The same starting point. Completely different destinations.

Have a look at Adelaide, Perth and Brisbane. Their bubbles are large and deep red. Both cities have seen enormous price growth since Q1 2018, meaning a house that cost $500,000 in early 2018 would now sell for well over $1M.

On the other hand, Melbourne’s bubble seems relatively small. It has barely moved. Sydney sits in between: it surged after COVID, then gave back much of those gains when interest rates rose sharply in 2022.

But what does this mean in real life? For someone who already owned a Brisbane house in 2018, it represents $400,000-$600,000 in equity. For someone who did not, e.g., a renter, a younger worker, someone who arrived in Brisbane after the surge, it represents the amount by which the entry price has moved out of reach. The same number reads as a windfall or a locked door depending on which side of 2018 you were on.

When did Brisbane actually overtake Melbourne?

This index map shows how each city’s median house price has changed relative to its own 2018 baseline. While useful for comparing growth rates, it obscures the underlying price gap between Brisbane and Melbourne and how that relationship evolved over time.

To reveal that story, we switch to raw median house prices in actual dollars, plotted quarter by quarter from 2018 to today. This view makes it possible to see how major events, including the COVID-era housing boom and subsequent RBA interest rate rises, shaped the trajectory of both markets and ultimately led to Brisbane overtaking Melbourne in 2026.

The chart shows median house prices in Brisbane and Melbourne from 2018 to mid-2026. Brisbane held a significant affordability advantage for most of the period, but two macro shocks eroded Brisbane’s relative affordability: COVID-era population movement drove sustained demand into Southeast Queensland as remote work made the region newly accessible, while Melbourne’s market softened under prolonged lockdowns; then the RBA’s 2022 rate hiking cycle hit Melbourne’s more leveraged market harder. By early 2025 Brisbane had lost its price advantage over Melbourne for the first time, and by mid-2026 sits roughly $200K more expensive at ~$1.4M, a historically significant shift for Brisbane home buyers.

Buyers priced out almost everywhere

Australia’s housing affordability crisis did not begin with COVID, but the pandemic years accelerated it dramatically and spread it almost everywhere. Since 2018, no major market has been spared.

The picture is unambiguous: buyers who entered the market before COVID saw their wealth grow substantially depending on where they bought. Those who did not are now facing a market that is structurally harder to enter in every city and region shown, with no meaningful affordability refuge left anywhere on the map.

Every market has seen dramatic price growth, with Brisbane, Adelaide, and the Gold Coast and Sunshine Coast surging well above their 2018 baseline, effectively doubling or more the cost of entry for buyers. Sydney and Canberra sit significantly higher, while even Melbourne, the weakest performer, has still climbed meaningfully. The COVID period accelerated the erosion sharply, as remote work drove demand into lifestyle and regional markets that had previously offered relative relief. The RBA’s 2022 rate hikes provided only a brief pause before prices resumed climbing. For anyone who did not own property before COVID, the window has narrowed in every direction, with no major Australian market offering a meaningful affordability alternative.

It’s not just about houses.

The signal investors watch but headlines ignore: rental yield.

Here is an under-reported fact about the Melbourne market.

What is rental yield? Rental yield is the annual rent a property earns, expressed as a percentage of its value. A $1,000,000 property that earns $40,000 per year in rent has a gross yield of 4%.

Yield and price can move in opposite directions. When prices rise fast but rents do not keep up, yield falls (you are paying more for the same income stream). When prices stagnate but rents keep rising, yield climbs (the same asset now produces a higher return relative to its cost).

As prices flatlined after 2022, rents kept rising in Melbourne, driven by a post-COVID return of students and renters to the city. The result: Melbourne’s gross rental yield has been quietly climbing back, making it increasingly attractive to investors who care about income rather than capital growth.

Perth shows the mirror image. Prices surged so fast that yields compressed dramatically, even as strong demand kept rents rising. The capital growth story crowded out the income story.

At some point, rising Melbourne yields will bring income-motivated buyers back into the market. That is what sets a floor under prices. The chart below shows where each city’s yield is heading.

This chart reveals how Melbourne’s rising yield is a quiet signal that the flat-price period may be nearing its floor. Perth’s falling yield reflects a market where price growth has outpaced rental income, a dynamic that historically precedes a slowdown. One more thread: the sea-changers.

The Sunshine Coast and Gold Coast did not follow the economic fundamentals story. They followed people.

What is a sea-changer? A sea-changer is someone who leaves a major city for a coastal or regional lifestyle (typically trading a longer commute, or none at all, for better quality of life, more space, and a lower cost of living). Before 2020, this was a slow trickle. COVID changed it almost overnight.

When offices closed and remote work became the norm, many households discovered they no longer needed to live within commuting distance of a capital city. The Sunshine Coast (roughly one hour north of Brisbane by car) and the Gold Coast (roughly one hour south) suddenly became viable full-time homes for people whose salaries were still priced at Sydney or Melbourne levels. The combination of high-city incomes and coastal-town prices created intense competition for local properties.

The result in the data is a sharp, sudden spike from mid-2020 to mid-2022, unlike anything these markets had seen before. The Sunshine Coast, in particular, moved from near the national median to well above it within 18 months.

The open question is in the 2023-2026 segment: did that demand hold, or is it unwinding as employers pull people back to offices? Brisbane is shown as context. It is geographically close but driven by entirely different forces (structural population growth and infrastructure investment from the upcoming Olympics). Separating the coastal spike from the broader southeast Queensland story is important for understanding how durable each is.

The COVID migration spike is visible and dramatic in the data, far exceeding the trajectory of any prior cycle. But the post-2022 segment is the more telling signal. Markets holding their gains above Brisbane’s trend line have sticky demand (people who moved and stayed). Markets converging back toward it are showing signs of unwinding as return-to-office pressures mount. Whether the sea-change gains are permanent or a temporary dislocation is one of the most important unresolved questions in Australian property right now.

Chapter 2: A Magnifier on Melbourne’s Market

The Domain API also gives us a more fine-grained lens over individual cities. Consider a market where expensive suburbs are softening while affordable ones are rising. The city-level median would look flat. But the story underneath would be a market actively splitting in two. That is Melbourne.

The suburb performance statistics endpoint used in Chapter 1 is powerful precisely because of its breadth: a handful of API calls return a decade of data across every major city. Its limitation, however, is fundamental. It reports what a whole suburb did on average, aggregated over time. It cannot reveal what is happening at different ends of the price spectrum within that suburb, or whether those ends are moving in opposite directions across the same period.

The statistics endpoint told us Melbourne is flat. What it cannot tell us is that “flat” may be the average of two opposite movements, one at the expensive end and one at the affordable end, cancelling each other out.

To look inside Melbourne’s flatness, we shift to a different kind of data: individual property transactions. Every sold listing carries a price, date, suburb, property type, and bedroom count. This is far more granular than suburb-level aggregates, and far more expensive to collect, but it is the only way to test whether Melbourne’s flat median represents a genuine pause or an artefact of averaging two very different markets into one number.

The map below plots every sold property transaction in our dataset for two Melbourne suburbs: Hawthorn (pink-purple dots) and Frankston (blue dots). Each dot is a real sale. Darker colours indicate higher prices within each suburb’s own price range.

Before diving into the numbers, notice how differently the two suburbs sit on the map. Hawthorn’s transactions cluster tightly in a small, dense patch of inner Melbourne. Frankston’s spread across a much wider footprint at the city’s edge. The geography itself signals different buyer types: professionals competing for scarce inner-city homes in one case, families and first-home buyers seeking space and affordability in the other.

Hawthorn sits 6km east of the Melbourne CBD. Leafy streets, Victorian terraces, elite schools. The suburb is generally labelled with high socioeconomic status and established heritage enclave. Buyers are established professionals and upsizers competing for a constrained supply of character homes. The median house price was $2.9M in Q1 2018. This is a market shaped by sentiment and interest rates. Since Q1 2018, it is down around 19%.

Hawthorn and Frankston represent two different relationships with interest rates, income levels, and borrowing capacity. Chapter 3: Other Market Indicators

Price is only the headline. Underneath it sit a handful of quieter indicators that often move first, and frequently tell a richer story than the median ever could. How long a property takes to sell, whether vendors feel confident enough to risk an auction, and which kinds of homes are actually changing hands all shift before, or independently of, the sold price.

This final part puts three of those indicators side by side for Hawthorn and Frankston: days on market, the auction-versus-private-treaty mix, and the bedroom composition of sales. Each is drawn from the same Domain endpoints used earlier, and each adds a dimension the price charts alone cannot capture, from early-warning timing signals to whether a trend is broad-based or driven by a shift in what is being sold.

Before prices move, time moves.

Days on market is the market’s early warning system.

What is days on market? Days on market (DOM) measures how long a property sits listed for sale before a buyer commits. In a hot market, properties sell quickly: buyers compete, the best offer wins within days, and vendors rarely need to negotiate on price. When demand softens, properties linger. Vendors hold out for their asking price. Buyers become selective. The DOM climbs. And critically, it climbs before the sold price data shows any movement, because price indices are backward-looking (they only capture deals that have already closed), while DOM shows you what is happening right now.

This chart shows how Hawthorn’s days on market have been trending upward, i.e., properties are taking longer to sell. Frankston’s have held steady or declined, reflecting persistent buyer demand. In a market where prices often move slowly and with a lag, DOM is the leading indicator: Hawthorn’s lengthening time-on-market began climbing several quarters before its sold prices started to soften. If you were watching DOM closely, you had advance warning of what the price data would confirm later.

How you sell and which market you are in.

In a strong market, vendors run auctions. Buyers compete, bids exceed reserve, and sale is unconditional. The vendor holds all the cards.

When confidence softens, vendors shift to private treaty. They test the market quietly, avoid a public pass-in, and negotiate terms. It is a rational response to demand thinning at the top.

The Domain API returns three sale method values for sold properties:SoldByAuction: Cleared under the hammer - SoldPriorToAuction: Sold before auction day (a signal of strong buyer intent, or vendor caution) - SoldByPrivateTreaty: Negotiated sale

This chart shows the auction share over time. Hawthorn shows a clear shift away from auction-dominant selling in the post-2022 period. Auctions require buyer confidence: they are public, unconditional, and competitive. When that confidence softens (because rate rises have thinned the pool of buyers who can bid at premium prices), vendors switch to private treaty to avoid a public pass-in (a failed auction where no bid reached the vendor’s reserve, which signals weakness and undermines future negotiations). The growing share of private treaty sales in Hawthorn is consistent with a market where buyer appetite at the premium end has thinned. Frankston’s mix tells the opposite story: private treaty is the structural default for first-home buyers (who need conditional contracts to arrange finance), and its mix has remained stable throughout the cycle, reflecting unchanged demand fundamentals.

What if we could zoom in even further?

Breaking the data down by bedroom count lets us test something the median alone cannot: whether what is selling has changed, not just what it is selling for. If Hawthorn’s softening is driven by fewer large prestige homes transacting rather than by genuine price falls across the board, that is a composition effect, not a price signal. The four panels work as two pairs. The top row shows sales volume by bedroom tier over time; the bottom row shows median sold price for each tier. Reading them together tells you whether a shift in the mix of transactions is distorting the overall trend, or whether prices are genuinely moving for every property type.

In Hawthorn, both things are happening at once. The 4 and 5-bedroom segment has become less active as a share of sales, and the prices of those larger properties have also softened, particularly after 2022. These are precisely the homes most sensitive to rate rises: large, expensive, typically bought by upsizers carrying significant mortgage debt. Smaller Hawthorn properties have held up comparatively better, which partially masks how much pressure the premium end is under. Frankston tells a different story. The market is overwhelmingly 3-bedroom houses throughout the entire study period, with no meaningful composition shift. The price growth visible in the earlier chart is not a statistical artefact. It is uniform across bedroom tiers and buyer profiles, meaning Frankston’s gains are real and broad-based. Every segment moving together is, in its own way, the less forgiving outcome for buyers who were hoping the affordable end might stay within reach.

Chapter 4: Conclusion

What the data showed

Three things held across every chart in this notebook.

The affordability shift since 2018 is both universal and structural. Not a single major Australian market has been spared. Cities, coastal towns, houses, and units have all moved in the same direction. The entry cost to the Australian property market is now roughly double what it was seven years ago, and there is no geography left as an accessible alternative. The data reads as a windfall or a locked door depending entirely on which side of 2018 you were sitting, and the locked-door reading is now the majority experience.

City-level medians conceal as much as they reveal. Melbourne’s headline flatness looked like stability until suburb-level and then transaction-level data showed what was underneath: a premium inner-city market softening under rate pressure at the same time as an outer-ring affordable market was rising hard. The aggregate median averaged those two stories into an apparent non-event. Anyone relying on city-level numbers alone would have missed both movements entirely.

Sold price is a lagging indicator. Days on market in Hawthorn began climbing several quarters before the sold price data reflected any softening. The shift in sale method mix, from auctions toward private treaty, confirmed the direction of travel before prices followed. Both signals were readable from the same API calls that produced the price data. The tools needed to lead the market, rather than simply follow it, are already embedded in the data.

Two tools, two types of question

The Domain API via AURIN makes available two fundamentally different data products, and this notebook used both.

The suburb performance statistics endpoint is built for breadth. A handful of API calls return a decade of quarterly data for every major city, making it possible to map national divergence, compare growth trajectories, and track rental yield with minimal code. The trade-off is resolution: the endpoint reports what a suburb did on average, and the distribution of individual transactions underneath that average is invisible to it.

The residential listings endpoint is built for depth. Each sold listing carries a price, date, suburb, property type, bedroom count, and coordinates. That granularity is what made the Hawthorn vs Frankston analysis possible: it is the only way to test whether a city-level trend is genuine, or an artefact of averaging two opposite movements into a single number. The cost is higher (more API calls, more data handling, a meaningful credit budget), but some questions cannot be answered any other way.

Knowing which data fits the question is half the work. The statistics endpoint is the right starting point for almost every housing research project. Transaction-level listings become necessary when aggregate data raises a question that aggregation itself cannot answer.

What this enables for housing research

The two endpoints work together to support a research sequence that is hard to replicate with traditional datasets: start with a national or city-level question using suburb statistics, identify an anomaly, then pull transaction-level data to look inside it. This is exactly what this notebook did, moving from the national divergence story down to individual streets in two Melbourne suburbs.

What makes that sequence practical for researchers is scale and speed. A city’s ten-year quarterly price history arrives in a single API call. Thousands of individual property transactions can be pulled in a script that runs overnight. The data refreshes as new sales settle and are reported to Domain, meaning the same workflow that produced this notebook can be run again in six months and will return a more current picture without any change to the code.

# ── Manifest helpers (resume-safe: already-downloaded files are skipped) ────

def _load_manifest():
    return json.loads(_MANIFEST.read_text()) if _MANIFEST.exists() else {"captures": []}

def _save_manifest(m):
    _MANIFEST.write_text(json.dumps(m, indent=2))

def _already_captured(m, rel):
    return str(rel) in {e["file"] for e in m["captures"]}

def _record_capture(m, rel, endpoint, params, status, section, empty=False):
    m["captures"].append({
        "file":             str(rel),
        "endpoint":         endpoint,
        "params":           params,
        "captured_at":      datetime.now(timezone.utc).isoformat(),
        "http_status":      status,
        "empty_response":   empty,
        "datatale_section": section,
    })
    _save_manifest(m)

def _save_json(rel, data):
    dest = _DATATALE / rel
    dest.parent.mkdir(parents=True, exist_ok=True)
    dest.write_text(json.dumps(data, indent=2))

def _save_probe(rel, response):
    _save_json(rel, {
        "x_total_count": int(response.headers.get("X-Total-Count", 0)),
        "body":          response.json() if response.text.strip() else {},
    })

# The Domain API caps results at pageNumber * pageSize <= 1000. This paginator

def _extract_listings(data):
    if not isinstance(data, list):
        return []
    out = []
    for item in data:
        raw = item.get("listing") or item.get("listings")
        if isinstance(raw, dict):
            out.append(raw)
        elif isinstance(raw, list):
            out.extend(raw)
    return out


def _fetch_listings_pages(tracker, manifest, suburb, postcode, state,
                          start_date, out_dir, datatale_section):
    out_abs    = _DATATALE / out_dir
    state_file = out_abs / "fetch_state.json"
    page_size  = 200

    if state_file.exists():
        st       = json.loads(state_file.read_text())
        cursor   = st["cursor"]
        run      = st["run"]
        seen_ids = set(st["seen_ids"])
        print(f"  Resuming: cursor={cursor}  completed runs={run}  seen={len(seen_ids)}")
    else:
        cursor   = start_date
        run      = 0
        seen_ids = set()

    probe_path = f"{out_dir}/probe.json"
    if not _already_captured(manifest, probe_path):
        probe_r = tracker.post(
            f"{PROXY_BASE}/v1/listings/residential/_search",
            json_body={
                "listingType": "Sold",
                "listedSince": start_date,
                "sort": {"sortKey": "DateListed", "direction": "Ascending"},
                "locations": [{"state": state, "suburb": suburb,
                               "postCode": postcode, "includeSurroundingSuburbs": False}],
                "pageSize": 1, "pageNumber": 1,
            },
        )
        _save_probe(probe_path, probe_r)
        total_count = int(probe_r.headers.get("X-Total-Count", 0))
        _record_capture(manifest, probe_path,
                        endpoint="POST /v1/listings/residential/_search",
                        params={"suburb": suburb, "postcode": postcode,
                                "listedSince": start_date},
                        status=probe_r.status_code,
                        section=datatale_section)
        print(f"  Probe: {total_count:,} total records for {suburb}")
    else:
        probe_data  = json.loads((_DATATALE / probe_path).read_text())
        total_count = probe_data.get("x_total_count", 0)
        print(f"  Probe already captured: {total_count:,} records")

    if total_count == 0:
        print(f"  WARNING: probe returned 0 -- check suburb name and postcode")
        return

    payload = {
        "listingType": "Sold",
        "sort":        {"sortKey": "DateListed", "direction": "Ascending"},
        "locations":   [{"state": state, "suburb": suburb,
                          "postCode": postcode, "includeSurroundingSuburbs": False}],
    }

    while True:
        run      += 1
        page      = 1
        run_batch = []
        hit_limit = False
        fetch_err = False

        while True:
            page_path = f"{out_dir}/run_{run:03d}_page_{page:03d}.json"

            if _already_captured(manifest, page_path):
                existing = json.loads((_DATATALE / page_path).read_text())
                run_batch.extend(_extract_listings(existing))
                print(f"    run {run:03d} page {page:03d} -- already captured, skip")
                page += 1
                continue

            r = tracker.post(
                f"{PROXY_BASE}/v1/listings/residential/_search",
                json_body={**payload, "listedSince": cursor,
                           "pageSize": page_size, "pageNumber": page},
            )

            if r.status_code == 400:
                hit_limit = True
                break
            if r.status_code != 200:
                print(f"    ERROR {r.status_code} run {run:03d} page {page:03d}: {r.text[:100]}")
                print(f"  Stopping -- re-run cell to resume from cursor {cursor}")
                fetch_err = True
                break

            data     = r.json()
            listings = _extract_listings(data)
            new      = [l for l in listings if l.get("id") and l["id"] not in seen_ids]

            _save_json(page_path, data)
            _record_capture(manifest, page_path,
                            endpoint="POST /v1/listings/residential/_search",
                            params={**payload, "listedSince": cursor,
                                    "pageSize": page_size, "pageNumber": page},
                            status=r.status_code,
                            section=datatale_section)
            run_batch.extend(listings)
            seen_ids.update(l["id"] for l in listings if l.get("id"))
            print(f"    run {run:03d} page {page:03d}: {len(listings)} records  "
                  f"({len(new)} new  {len(seen_ids)} unique total)  [{tracker.total} credits]")

            if len(listings) < page_size:
                break
            page += 1

        if fetch_err:
            break

        if not hit_limit:
            print(f"  Done. {len(seen_ids):,} unique listings for {suburb}.")
            break

        if not run_batch:
            print("  Empty batch after 400 -- stopping")
            break

        dates = [l.get("dateListed", "") for l in run_batch if l.get("dateListed")]
        if not dates:
            print("  No dateListed in batch -- cannot advance cursor, stopping")
            break
        new_cursor = max(dates)[:10]
        if new_cursor == cursor:
            print(f"  Cursor stuck at {cursor} (all same date) -- stopping")
            break

        print(f"  Limit hit. Advancing cursor {cursor} -> {new_cursor}")
        cursor = new_cursor
        state_file.write_text(json.dumps(
            {"cursor": cursor, "run": run, "seen_ids": list(seen_ids)}, indent=2,
        ))

            "meanRentYield":   r["values"].get("meanRentYield"),
        }
        for r in rows
        if r["values"].get("medianSoldPrice")   # skip quarters with no price data
    ]
    if not records:
        return None
    return pd.DataFrame(records).set_index("date").sort_index()


def pct_change_from_q1_2018(df, col="medianSoldPrice"):
    """Add an 'indexed' column expressing price as % change from the Q1 2018 level.

    Q1 2018 is defined as any record where year==2018 and month in {1, 2, 3}.
    The baseline is the first available price in that window.

    Examples:
        indexed = +80  ->  price is 80% above its Q1 2018 level
        indexed =  -5  ->  price is 5% below its Q1 2018 level
        indexed =   0  ->  price equals its Q1 2018 level (the baseline quarter)

    Returns df unchanged (without an 'indexed' column) if no Q1 2018 data exists.
    """
    mask     = (df.index.year == 2018) & (df.index.month.isin([1, 2, 3]))
    baseline = df.loc[mask, col].dropna()
    if baseline.empty:
        return df
    df = df.copy()
    df["indexed"] = (df[col] / baseline.iloc[0] - 1) * 100   # 0 = Q1 2018 level
    return df

# Melbourne property transaction map: Hawthorn (inner, premium) vs Frankston (outer, affordable)
# Each dot is a sold property. Toggle layers with the control in the top-right corner.

# Load listings first (needed for lat/lon)
def load_listings_dir(dir_path, study_start=STUDY_START, study_end=STUDY_END):
    """Load all run_*_page_*.json listing files from a directory into a single DataFrame.

    The Domain residential listings API uses 'soldData' (not 'soldDetails') for
    sold price and date, and 'propertyDetails' for type, bedrooms, and coordinates.

    Each JSON file contains a list of listing objects. Multiple files per suburb
    are expected (one per API page); all are concatenated and deduplicated by listing id.

    Columns returned:
        id, sold_date, sold_price, sale_method, property_type,
        bedrooms, lat, lon, quarter (Period), sold_year

    Filters applied:
        - sold_date within [study_start, study_end]
        - sold_price > $100,000 (removes data entry errors and non-arm's-length sales)
        - rows missing sold_date or sold_price are dropped
    """
    dir_path = pathlib.Path(dir_path)
    if not dir_path.exists():
        print(f"  WARNING: {dir_path} not found -- run Phase D first")
        return pd.DataFrame()
    records = []
    for f in sorted(dir_path.glob("run_*_page_*.json")):
        data = json.loads(f.read_text())
        if not isinstance(data, list):
            continue
        for item in data:
            lst = item.get("listing") or item.get("listings")
            if not isinstance(lst, dict):
                continue
            sold = lst.get("soldData") or {}
            prop = lst.get("propertyDetails") or {}
            records.append({
                "id":            lst.get("id"),
                "sold_date":     sold.get("soldDate"),
                "sold_price":    sold.get("soldPrice"),
                "sale_method":   sold.get("saleMethod", ""),
                "property_type": prop.get("propertyType", "unknown").lower(),
                "bedrooms":      prop.get("bedrooms"),
                "lat":           prop.get("latitude"),
                "lon":           prop.get("longitude"),
            })
    if not records:
        return pd.DataFrame()
    df = pd.DataFrame(records).drop_duplicates(subset="id")
    df["sold_date"]  = pd.to_datetime(df["sold_date"], errors="coerce").dt.tz_localize(None)
    df["sold_price"] = pd.to_numeric(df["sold_price"], errors="coerce")
    df["lat"]        = pd.to_numeric(df["lat"], errors="coerce")
    df["lon"]        = pd.to_numeric(df["lon"], errors="coerce")
    df = df.dropna(subset=["sold_date", "sold_price"])
    df = df[(df["sold_date"] >= study_start) & (df["sold_date"] <= study_end)]
    df = df[df["sold_price"] > 100_000]
    df["quarter"]    = df["sold_date"].dt.to_period("Q")
    df["sold_year"]  = df["sold_date"].dt.year
    return df.sort_values("sold_date").reset_index(drop=True)

# Build a folium map with two togglable suburb layers.
# Each dot represents one sold property, coloured by price within that suburb's own range
# (using a 5-step palette from light to dark). This makes price variation visible
# within each suburb without being distorted by the large price gap between the two suburbs.

# Maps a sold price to one of 5 hex colours: ratio 0 -> palette[0] (cheapest),
# ratio 1 -> palette[4] (most expensive). Clipped to [0, 1] so outliers beyond
# the 5th-95th percentile range don't overflow the index.
def price_to_colour(price, p_min, p_max, palette):
    ratio = min(1.0, max(0.0, (price - p_min) / max(1, p_max - p_min)))
    idx = min(4, int(ratio * 5))
    return palette[idx]

# Compute the [[south, west], [north, east]] bounding box of a GeoJSON polygon
# so the closeread narrative can fly the map to each suburb's boundary on scroll.
def geojson_bounds(geojson):
    lats, lons = [], []
    def walk(coords):
        if coords and isinstance(coords[0], (int, float)):
            lons.append(coords[0]); lats.append(coords[1])
        else:
            for c in coords:
                walk(c)
    for feat in geojson.get("features", [geojson]):
        walk(feat.get("geometry", feat)["coordinates"])
    return [[min(lats), min(lons)], [max(lats), max(lons)]]

(function () {
  var BOUNDS = __BOUNDS__;

  var home = null;
  function getMap() { return window[MAP_NAME]; }
  window.addEventListener('message', function (ev) {
    var d = ev.data || {};
    if (d.type !== 'cr-zoom') return;
    var map = getMap();
    if (!map) return;
    if (!home) home = { center: map.getCenter(), zoom: map.getZoom() };
    if (d.suburb && BOUNDS[d.suburb]) {
      map.flyToBounds(BOUNDS[d.suburb], { padding: [25, 25], duration: 1.1 });
    } else {
      map.flyTo(home.center, home.zoom, { duration: 1.1 });
    }
  });
}());
</script>
""".replace("__BOUNDS__", json.dumps(SUBURB_BOUNDS)).replace("__MAP_NAME__", m.get_name())
